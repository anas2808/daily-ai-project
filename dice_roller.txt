    <div class="dice-container">
        <div class="dice-face" id="diceFace">1</div>
        <button class="roll-button" id="rollButton">Roll Dice</button>
    </div>

    <script>
        /**
         * Function to roll a dice and return a random number between 1 and 6.
         * @returns {number} - A random integer between 1 and 6.
         */
        function rollDice() {
            return Math.floor(Math.random() * 6) + 1;
        }

        /**
         * Function to update the dice face on the UI with a new random number.
         */
        function updateDiceFace() {
            try {
                const diceFaceElement = document.getElementById('diceFace');
                const newRoll = rollDice();
                // Update the dice face element with the new roll value
                diceFaceElement.textContent = newRoll;
            } catch (error) {
                console.error("Error updating the dice face:", error);
            }
        }

        /**
         * Main execution block to initialize event listeners.
         */
        function main() {
            const rollButton = document.getElementById('rollButton');
            // Add click event listener to the roll button
            rollButton.addEventListener('click', updateDiceFace);
        }

        // Execute the main function once the DOM is fully loaded
        document.addEventListener('DOMContentLoaded', main);
    </script>
