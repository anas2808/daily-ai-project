        <div style="text-align: center;">
            <button class="nav-button" onclick="previousSlide()">Previous</button>
            <button class="nav-button" onclick="nextSlide()">Next</button>
        </div>
    </div>

    <script>
        // Initialize slide index
        let currentSlideIndex = 0;

        // Function to show the next slide
        function nextSlide() {
            try {
                const slides = document.getElementsByClassName('slide');
                slides[currentSlideIndex].classList.remove('active-slide');
                currentSlideIndex = (currentSlideIndex + 1) % slides.length;
                slides[currentSlideIndex].classList.add('active-slide');
            } catch (error) {
                console.error('Error navigating to the next slide:', error);
            }
        }

        // Function to show the previous slide
        function previousSlide() {
            try {
                const slides = document.getElementsByClassName('slide');
                slides[currentSlideIndex].classList.remove('active-slide');
                currentSlideIndex = (currentSlideIndex - 1 + slides.length) % slides.length;
                slides[currentSlideIndex].classList.add('active-slide');
            } catch (error) {
                console.error('Error navigating to the previous slide:', error);
            }
        }
    </script>
